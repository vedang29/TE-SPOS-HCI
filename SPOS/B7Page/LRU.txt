#include <iostream>
#include <iomanip>
using namespace std;

int main() 
{
    int n, framesCount;
    cout << "Enter number of pages in reference string: ";
    cin >> n;
    
    int ref[n];
    cout << "Enter reference string: ";
    for (int i = 0; i < n; i++)
        cin >> ref[i];

    cout << "Enter number of frames: ";
    cin >> framesCount;

    int frames[10], lastUsed[10];
    int hit = 0, miss = 0;

    for (int i = 0; i < framesCount; i++) 
    {
        frames[i] = -1;
        lastUsed[i] = -1;
    }

    cout << "\n      LRU Page Replacement      \n";
    cout << setw(10) << "Page" << setw(20) << "Frames" << setw(15) << "Result\n";
    cout << "\n";

    for (int i = 0; i < n; i++) 
    {
        bool found = false;
        for (int j = 0; j < framesCount; j++) 
        {
            if (frames[j] == ref[i]) 
            {
                hit++;
                lastUsed[j] = i;
                found = true;
                break;
            }
        }

        cout << setw(10) << ref[i] << setw(15);

        if (!found) 
        {
            miss++;
            int replaceIndex = -1;

            for (int j = 0; j < framesCount; j++) 
            {
                if (frames[j] == -1) 
                {
                    replaceIndex = j;
                    break;
                }
            }

            if (replaceIndex == -1) 
            {
                int min = 0;
                for (int j = 1; j < framesCount; j++)
                    if (lastUsed[j] < lastUsed[min])
                        min = j;
                replaceIndex = min;
            }

            frames[replaceIndex] = ref[i];
            lastUsed[replaceIndex] = i;

            for (int k = 0; k < framesCount; k++)
                (frames[k] == -1) ? cout << "- " : cout << frames[k] << " ";
            cout << setw(15) << "Miss\n";
        } 
        else 
        {
            for (int k = 0; k < framesCount; k++)
                (frames[k] == -1) ? cout << "- " : cout << frames[k] << " ";
            cout << setw(15) << "Hit\n";
        }
    }

    cout << "\n";
    cout << "Total Hits: " << hit << "\n";
    cout << "Total Misses (Page Faults): " << miss << "\n";
    cout << "Hit Ratio: " << fixed << setprecision(2) << (float)hit / n << "\n";
    cout << "Miss Ratio: " << fixed << setprecision(2) << (float)miss / n << "\n";

    return 0;
}
